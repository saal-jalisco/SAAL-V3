function generarCodigoSesion() { if (!textoActual) { alert('Primero selecciona un texto.'); return; } // Generar un código determinístico a partir del título del texto const tituloLimpio = textoActual.titulo .normalize('NFD') .replace(/[\u0300-\u036f]/g, '') .toUpperCase() .replace(/[^A-Z0-9]/g, ''); // Tomar las primeras 4 letras del título let codigo = tituloLimpio.substring(0, 4); // Si el código tiene menos de 4 caracteres, rellenar con números while (codigo.length < 4) { codigo += Math.floor(Math.random() * 10); } codigoSesion = codigo; document.getElementById('codigoSesion').textContent = codigoSesion; // Guardar el texto asociado al código en una variable global (no en localStorage) window.textoPorCodigo = { [codigoSesion]: textoActual }; alert('✅ Código de sesión generado: ' + codigoSesion + '\nCompártelo con el alumno.'); }
